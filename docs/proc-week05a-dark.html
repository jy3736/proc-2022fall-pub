課堂練習 1 - 以兩個互不干擾的函數，分別控制LED的閃爍。

Arduino Sketch

int ledR = 13;
int ledG = 8;

unsigned long t0 = millis();

unsigned long tR = t0;
unsigned long tG = t0;

unsigned long intR = 250;
unsigned long intG = 500;

bool staR = false;
bool staG = false;

#define HL(x) ((x)?HIGH:LOW)


void setup() {
  	pinMode(ledR, OUTPUT);
  	pinMode(ledG, OUTPUT);
}

void procA() {
	if (t0-tR<intR) return;
  	tR = t0;
  	staR = !staR;
  	digitalWrite(ledR,HL(staR));
}

void procB() {
	if (t0-tG<intG) return;
  	tG = t0;
  	staG = !staG;
  	digitalWrite(ledG,HL(staG));
}

void loop() {
  	t0 = millis();
    procA();
  	procB();
	delay(0);
}

課堂練習 2 - 優化改良【課堂練習 1】，合併控制LED的函數。

Arduino Sketch

#define NP 2
unsigned long t0 = millis();

int led[] = {13,8};
unsigned long tx[] = {t0,t0};
unsigned long intval[] = {250,500};
bool sta[] = {false,false};

#define HL(x) ((x)?HIGH:LOW)

void setup() {
  for(int i=0; i<NP; i++)
    pinMode(led[i], OUTPUT);
}

void procLED() {
  for(int i=0; i<NP; i++) {
	if (t0-tx[i]<intval[i]) return;
  	tx[i] = t0;
  	sta[i] = !sta[i];
  	digitalWrite(led[i],HL(sta[i]));
  }
}

void loop() {
  t0 = millis();
  procLED();
  delay(0);
}

課堂練習 4 - 擬平行(Pseudo Parallelism)狀態機(FSM)函數設計。

Arduino Sketch

#define HL(x) ((x)?HIGH:LOW)

const int R = 13;
const int Y = 10;
const int G = 8;

// R(2000),RY(500),G(3000),Y(500)
unsigned long intval[] = {2000,500,3000,500};

unsigned long t0 = millis();
unsigned long tx = t0;

int cs = 0;
unsigned long td = 0; 

void LED(bool r, bool y, bool g) {
  digitalWrite(R,HL(r));
  digitalWrite(Y,HL(y));
  digitalWrite(G,HL(g));  
}

void setup() {
  pinMode(R, OUTPUT);
  pinMode(Y, OUTPUT);
  pinMode(G, OUTPUT);
  LED(false,false,false);
}

void FSM() {
  if (t0-tx<td) return;
  tx = t0;
  td = intval[cs];
  switch(cs) {
  	case 0: LED(true,false,false);
    		cs = 1;
    		break;
  	case 1: LED(true,true,false);
    		cs = 2;
    		break;
  	case 2: LED(false,false,true);
    		cs = 3;
    		break;
  	case 3: LED(false,true,false);
    		cs = 0;
    		break;
    default: cs = 0;
  }
}

void loop() {
  t0 = millis();
  FSM();
  delay(0);
}

課堂作業 1 - 使用終端機命令，控制【課堂練習 4】交通號誌燈。

  • 將終端機處理程式，設計成擬平行(Pseudo Parallelism)函數。
  • 從終端機(serial)讀取命令。
  • 終端機命令定義：
    • 0 <t>：設定狀態0持續的時間，intval[0] = t。
    • 1 <t>：設定狀態1持續的時間，intval[1] = t。
    • 2 <t>：設定狀態2持續的時間，intval[2] = t。
    • 3 <t>：設定狀態3持續的時間，intval[3] = t。
    • 100：終止交通號誌燈循環。
    • 200：重置交通號誌燈。
    • 忽略其他的輸入組合。
  • 處理完終端機命令，自動重置交通號誌燈。
#define HL(x) ((x)?HIGH:LOW)

const int R = 13;
const int Y = 10;
const int G = 8;

// R(2000),RY(500),G(3000),Y(500)
unsigned long intval[] = {2000,500,3000,500};

unsigned long t0 = millis();
unsigned long tx = t0;

int cs = 0;
unsigned long td = 0; 

void LED(bool r, bool y, bool g) {
  digitalWrite(R,HL(r));
  digitalWrite(Y,HL(y));
  digitalWrite(G,HL(g));  
}

void setup() {
  Serial.begin(9600);
  pinMode(R, OUTPUT);
  pinMode(Y, OUTPUT);
  pinMode(G, OUTPUT);
  LED(false,false,false);
}

void FSM() {
  if (t0-tx<td) return;
  tx = t0;
  td = (cs<4)?intval[cs]:100;
  switch(cs) {
  	case 0: LED(true,false,false);
    		cs = 1;
    		break;
  	case 1: LED(true,true,false);
    		cs = 2;
    		break;
  	case 2: LED(false,false,true);
    		cs = 3;
    		break;
  	case 3: LED(false,true,false);
    		cs = 0;
    		break;
		// 增加第9個狀態，處理狀態機暫停。
    case 9: LED(false,false,false); 
				cs = 9;
    		break;
    default: cs = 0;
  }
}

void procSerial() {
	// 將Serial 終端機命令處理程式碼加在這個函數裡
}

void loop() {
  t0 = millis();
  procSerial();
  FSM();
  delay(0);
}
